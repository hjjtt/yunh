### 9.5.2 分布式事务处理

```java
@Service
public class OrderServiceImpl implements OrderService {

    @Autowired
    private CourseClient courseClient;

    @Autowired
    private RocketMQTemplate rocketMQTemplate;

    /**
     * 创建订单（使用事务消息保证最终一致性）
     */
    @GlobalTransactional
    @Override
    public OrderVO createOrder(CreateOrderDTO dto) {
        // 1. 创建订单（本地事务）
        Order order = new Order();
        order.setUserId(dto.getUserId());
        order.setCourseId(dto.getCourseId());
        order.setAmount(dto.getAmount());
        order.setStatus(OrderStatus.PENDING_PAY);
        orderMapper.insert(order);

        // 2. 发送事务消息（扣减库存）
        CourseStockMessage message = new CourseStockMessage();
        message.setCourseId(dto.getCourseId());
        message.setCount(1);
        message.setOrderId(order.getId());

        rocketMQTemplate.sendMessageInTransaction(
            "order_topic:stock_decrease",
            MessageBuilder.withPayload(message).build(),
            null
        );

        // 3. 发送订单创建消息（异步通知）
        OrderCreatedMessage createdMessage = new OrderCreatedMessage();
        createdMessage.setOrderId(order.getId());
        createdMessage.setUserId(dto.getUserId());
        createdMessage.setCourseId(dto.getCourseId());
        createdMessage.setAmount(dto.getAmount());

        rocketMQTemplate.convertAndSend("order_topic:order_created", createdMessage);

        return convertToVO(order);
    }
}
```

### 9.7.2 数据同步方案

```java
/**
 * 课程数据同步到 ES
 * 监听 RocketMQ 消息
 */
@Component
@RocketMQMessageListener(
    topic = "course_topic",
    selectorExpression = "course_updated",
    consumerGroup = "search-course-sync-group"
)
public class CourseSyncConsumer implements RocketMQListener<CourseMessage> {

    @Autowired
    private RestHighLevelClient esClient;

    @Override
    public void onMessage(CourseMessage message) {
        // 1. 获取课程完整数据
        CourseVO course = courseClient.getCourseById(message.getCourseId());

        // 2. 构建 ES 文档
        XContentBuilder builder = XContentFactory.jsonBuilder();
        builder.startObject()
            .field("id", course.getId())
            .field("title", course.getTitle())
            .field("subtitle", course.getSubtitle())
            .field("teacher_name", course.getTeacherName())
            .field("price", course.getPrice())
            .field("sales", course.getSales())
            .field("rating", course.getRating())
            .endObject();

        // 3. 索引到 ES
        IndexRequest request = new IndexRequest("yunh_course")
            .id(String.valueOf(course.getId()))
            .source(builder);

        esClient.index(request, RequestOptions.DEFAULT);
    }
}
```

## 步骤 9.8：数据统计

### 9.8.1 实时统计方案

```java
/**
 * 使用 Redis 进行实时统计
 */
@Service
public class StatisticsService {

    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    /**
     * 记录课程观看次数
     */
    public void recordCourseView(Long courseId) {
        String key = "statistics:course:view:" + LocalDate.now().format(DateTimeFormatter.ISO_DATE);
        redisTemplate.opsForHyperLogLog().add(key, String.valueOf(courseId));
        redisTemplate.opsForZSet().incrementScore("course:view:rank", String.valueOf(courseId), 1);
    }

    /**
     * 获取课程今日观看次数
     */
    public Long getCourseViewCount(Long courseId, LocalDate date) {
        String key = "statistics:course:view:" + date.format(DateTimeFormatter.ISO_DATE);
        return redisTemplate.opsForHyperLogLog().size(key);
    }

    /**
     * 获取课程排行
     */
    public Set<ZSetOperations.TypedTuple<String>> getCourseRank(int limit) {
        return redisTemplate.opsForZSet().reverseRangeWithScores("course:view:rank", 0, limit - 1);
    }
}
```

### 9.8.2 定时统计任务

```java
/**
 * 每日统计任务
 */
@Component
public class DailyStatisticsTask {

    @Autowired
    private StatisticsMapper statisticsMapper;

    /**
     * 每天凌晨 1 点执行
     */
    @Scheduled(cron = "0 0 1 * * ?")
    public void dailyStatistics() {
        LocalDate yesterday = LocalDate.now().minusDays(1);

        // 1. 统计昨日新增用户
        int newUserCount = statisticsMapper.countNewUsers(yesterday);

        // 2. 统计昨日订单数量
        int orderCount = statisticsMapper.countOrders(yesterday);

        // 3. 统计昨日销售额
        BigDecimal revenue = statisticsMapper.sumRevenue(yesterday);

        // 4. 统计昨日课程观看次数
        int viewCount = statisticsMapper.countCourseViews(yesterday);

        // 5. 保存统计结果
        DailyStatistics statistics = new DailyStatistics();
        statistics.setDate(yesterday);
        statistics.setNewUserCount(newUserCount);
        statistics.setOrderCount(orderCount);
        statistics.setRevenue(revenue);
        statistics.setViewCount(viewCount);

        statisticsMapper.insert(statistics);
    }
}
```
